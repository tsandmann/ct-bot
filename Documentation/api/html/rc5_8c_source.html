00300 void rc5_control(void) {
00301     static uint16_t RC5_Last_Toggle = 1; 
00302     uint16_t rc5 = ir_read(&rc5_ir_data); // empfangenes RC5-Kommando
00303 
00304     if (rc5 != 0) {
00305         /* Toggle kommt nicht im Simulator, immer gewechseltes Toggle-Bit sicherstellen */
00306 #ifdef PC
00307         RC5_Last_Toggle = (uint16_t) (!(rc5 & RC5_TOGGLE));
00308 #endif
00309         /* Bei Aenderung des Toggle-Bits, entspricht neuem Tastendruck, gehts nur weiter */
00310         if ((rc5 & RC5_TOGGLE) != RC5_Last_Toggle) { // Nur Toggle-Bit abfragen, bei Ungleichheit weiter
00311             RC5_Last_Toggle = rc5 & RC5_TOGGLE; // Toggle-Bit neu belegen
00312             RC5_Code = rc5 & RC5_MASK; // alle uninteressanten Bits ausblenden
00313         }
00314     }
00315 #ifndef DISPLAY_AVAILABLE
00316     /* Falls Display aus ist, ist auch GUI aus => Tastenbehandlung hier abarbeiten */
00317     if (RC5_Code != 0) {
00318         default_key_handler();
00319         RC5_Code = 0; // fertig, RC5-Puffer loeschen
00320     }
00321 #endif // ! DISPLAY_AVAILABLE
00322 }
