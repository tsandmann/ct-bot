00045     switch (state) {

00066             /* Map-Update abwarten */

00070         {
00071             state = 3;
00072         }
00073         break;
00074     case 3:
00075         BLOCK_BEHAVIOUR(data, 200);
00076         os_calc_utilization();
00077         BLOCK_BEHAVIOUR(data, 200);
00078         os_calc_utilization();
