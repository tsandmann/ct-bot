00064 void bot_classify_objects_behaviour(Behaviour_t * data) {
00065     static uint8_t measure_count;
00066     static int16_t object_brightness;
00067 
00068     switch (state) {
00069     case CO_SEARCH:
00070         /* Objekte suchen */
00071         state = CO_CATCH;
00072         /* Such-Algo / Explore-Verhalten */
00073         //break;
00074 
00075     case CO_CATCH:
00076         /* Objekt einfangen */
00077         bot_catch_pillar_turn(data, 180);
00078         state = CO_IDENTIFY;
00079         measure_count = 0;
00080         object_brightness = 0;
00081         break;
00082 
00083     case CO_IDENTIFY:
00084         /* Objekterkennung */
00085         if (data->subResult == BEHAVIOUR_SUBFAIL) {
00086             state = CO_END;
00087             bot_turn(data, -180);
00088             break;
00089         }
00090         if (measure_count < 10) {
00091             /* Durchschnitt aus 10 Messungen */
00092             object_brightness += sensLineL + sensLineR;
00093             measure_count++;
00094         } else {
00095             /* Identifizierung */
00096             object_brightness /= 20;
00097             state = CO_UNLOAD;
00098             uint8_t i;
00099             LOG_DEBUG("object_brightness=0x%x", object_brightness);
00100             /* Klasse suchen */
00101             for (i=0; i<sizeof(targets)/sizeof(int16_t)/3; i++) {
00102                 if (object_brightness < targets[i].treshold) {
00103                     LOG_DEBUG("object_class=%u", i);
00104                     targets[i].y += 100;
00105                     bot_goto_pos(data, targets[i].x, targets[i].y - 100, 270);
00106                     break;
00107                 }
00108             }
00109         }
00110         break;
00111 
00112     case CO_UNLOAD:
00113         /* Objekt ausladen */
00114         bot_unload_pillar(data);
00115         state = CO_HOME;
00116         break;
00117 
00118     case CO_HOME:
00119         /* zurueck zum Startpunkt */
00120         bot_goto_pos(data, 0, 0, 0);
00121         state = CO_SEARCH;
00122         break;
00123 
00124     default:
00125         exit_behaviour(data, BEHAVIOUR_SUBFAIL);
00126         return;
00127     }
00128 }
