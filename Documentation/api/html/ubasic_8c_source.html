00310 {

00312   int op;
00313 
00314   f1 = factor();

00317         op == TOKENIZER_SLASH ||
00318         op == TOKENIZER_MOD   ||
00319         op == TOKENIZER_MOD2) {
00320     tokenizer_next();
00321     f2 = factor();
00322     switch(op) {
00323     case TOKENIZER_ASTR:
00324       f1 = f1 * f2;
00325       break;
00326     case TOKENIZER_SLASH:
00327       f1 = f1 / f2;
00328       break;
00329     case TOKENIZER_MOD2:
00330     case TOKENIZER_MOD:
00331       f1 = f1 % f2;
00332       break;
00333     }
00334     op = tokenizer_token();
00335   }
00336   return f1;

00597     lop = tokenizer_token();
00598     while (lop == TOKENIZER_LOGAND ||
00599            lop == TOKENIZER_LOGOR)    {
00600         tokenizer_next();
00601 #if UBASIC_STRING
00602         if (ubasic_is_strexpr())
00603             r1 = strrelation();
00604         else
00605 #endif
00606             r1 = relation();
00607         if (lop == TOKENIZER_LOGAND)
00608             r = r & r1;
00609         else if (lop == TOKENIZER_LOGOR)
00610             r = r | r1;
00611         lop = tokenizer_token();
00612     }
00613 
00614         // Kurzform (IF ohne THEN/ELSE)?
00615     if (tokenizer_token() == TOKENIZER_THEN) accept(TOKENIZER_THEN); else no_then=1;
00616     if(r) {
00617         statement();
00618         // bei Kurzform darf kein ELSE kommen!
00619         if (no_then && (tokenizer_token() == TOKENIZER_ELSE)) {
00620             tokenizer_error_print(current_linenum, SHORT_IF_WITH_ELSE);
00621             ubasic_break();
00622         }
00623         // hmm..., hier ist man schon ein Token zu weit...
00624         if  ((tokenizer_token() == TOKENIZER_NUMBER)||
00625             (tokenizer_token() != TOKENIZER_ELSE)) return;
00626         jump_to_next_linenum();
00627     } else {
00628         do {
00629             tokenizer_next();
00630         } while(tokenizer_token() != TOKENIZER_ELSE &&
00631                 tokenizer_token() != TOKENIZER_CR &&
00632                 tokenizer_token() != TOKENIZER_ENDOFINPUT);
00633         if(tokenizer_token() == TOKENIZER_ELSE) {
00634             // bei Kurzform darf kein ELSE kommen!
00635             if (no_then) {
00636                 tokenizer_error_print(current_linenum, SHORT_IF_WITH_ELSE);
00637                 ubasic_break();
00638             }
00639             tokenizer_next();
00640             statement();
00641         } else  if(tokenizer_token() == TOKENIZER_CR) tokenizer_next();
00642     }

00683         jump_to_next_linenum();
00684     } else
00685 #else
00686     if (tokenizer_token() == TOKENIZER_STRING) {
00687         tokenizer_error_print(current_linenum, GOSUB_NO_EXT_SUBPROC);
00688         ubasic_break();
00689     } else
00690 #endif
00691     {
00692         linenum = expr();
00693         // es muss bis zum Zeilenende gelesen werden, um die Rueck-
00694         // sprungzeile fuer return zu ermitteln
00695         if (tokenizer_token() != TOKENIZER_CR) jump_to_next_linenum();
00696         else  tokenizer_next();
00697         tokenizer_next();
00698     }
00699     if(gosub_stack_ptr < MAX_GOSUB_STACK_DEPTH) {
00700         gosub_stack[gosub_stack_ptr].p_ptr = get_prog_text_pointer();
00701         #if UBASIC_EXT_PROC
00702             strncpy(gosub_stack[gosub_stack_ptr].p_name, current_proc, MAX_PROG_NAME_LEN);
00703         #endif
00704         gosub_stack_ptr++;
00705 #if UBASIC_EXT_PROC
00706         if (p_name[0]) {
00707             switch_proc(p_name);
00708         } else
00709 #endif
00710         jump_linenum(linenum);
00711     } else {
00712         tokenizer_error_print(current_linenum, GOSUB_STACK_DETH);
00713         ubasic_break();
00714     }
00715 }

00736 static void
00737 next_statement(void)
00738 {
00739   struct varinfo_t var;
00740 
00741   accept(TOKENIZER_NEXT);
00742   accept(TOKENIZER_VARIABLE);
00743   var = ubasic_get_varinfo();
00744   if(for_stack_ptr > 0 && var.varnum == for_stack[for_stack_ptr - 1].for_variable) {
00745     ubasic_set_variable(var, ubasic_get_variable(var) + for_stack[for_stack_ptr - 1].step);
00746     if(((ubasic_get_variable(var) <= for_stack[for_stack_ptr - 1].to) && !for_stack[for_stack_ptr - 1].downto)||
00747        ((ubasic_get_variable(var) >= for_stack[for_stack_ptr - 1].to) && for_stack[for_stack_ptr - 1].downto)
00748       ) {
00749       jump_to_prog_text_pointer(for_stack[for_stack_ptr - 1].next_line_ptr);
00750     } else {
00751       for_stack_ptr--;
00752       accept(TOKENIZER_CR);
00753     }
00754   } else {
00755     accept(TOKENIZER_CR);
00756   }
00757 
00758 }
00759 /*---------------------------------------------------------------------------*/
00760 static void
00761 for_statement(void)
00762 {
00763   int for_variable, to, step;
00764   unsigned char downto = 0;
00765   struct varinfo_t var;
00766 
00767   accept(TOKENIZER_FOR);
00768   accept(TOKENIZER_VARIABLE);
00769   var = ubasic_get_varinfo();
00770   for_variable = var.varnum;
00771   accept(TOKENIZER_EQ);
00772   ubasic_set_variable(var, expr());
00773   if (tokenizer_token() == TOKENIZER_TO) {
00774     downto = 0;
00775   } else if (tokenizer_token() == TOKENIZER_DOWNTO) {
00776     downto = 1;
00777   } else tokenizer_error_print(current_linenum, FOR_WITHOUT_TO);
00778   tokenizer_next();
00779   to = expr();
00780   if(tokenizer_token() == TOKENIZER_STEP) {
00781     tokenizer_next();
00782     step = expr();
00783   } else {
00784     step = 1;
00785   }
00786   if (downto) step *= -1;
00787   accept(TOKENIZER_CR);
00788   if(for_stack_ptr < MAX_FOR_STACK_DEPTH) {
00789     for_stack[for_stack_ptr].next_line_ptr = get_prog_text_pointer();
00790     for_stack[for_stack_ptr].for_variable = for_variable;
00791     for_stack[for_stack_ptr].to = to;
00792     for_stack[for_stack_ptr].step = step;
00793     for_stack[for_stack_ptr].downto = downto;
00794     for_stack_ptr++;
00795   } else {
00796     tokenizer_error_print(current_linenum, FOR_STACK_DETH);
00797     ubasic_break();
00798   }
00799 }

00959 static int ubasic_next_data(void) {
00960     struct tokenizer_pos_t save_pos;
00961     int val=0;
00962     // aktuelle Tokenizer-Position merken
00963     save_pos=tokenizer_get_position();

00982             //...
00983         } else if (tokenizer_token() == TOKENIZER_MINUS) {

00990             }
00991             // last_num im tokenizer manipulieren...
00992             tokenizer_set_num(tokenizer_num()*(-1));
00993 #if UBASIC_STRING
00994         } else if (tokenizer_token() == TOKENIZER_STRING) {
00995             //...
00996 #endif
00997         } else
00998             // unzulaessiges Token in DATA-Anweisung
00999             break;
01000     } while (tokenizer_token() != TOKENIZER_NUMBER &&
01001 #if UBASIC_STRING
01002              tokenizer_token() != TOKENIZER_STRING &&
01003 #endif
01004              tokenizer_token() != TOKENIZER_ENDOFINPUT &&
01005              !tokenizer_finished());
01006     // "ausreichend" DATA-Werte vorhanden...?
01007     if (tokenizer_token() == TOKENIZER_ENDOFINPUT || tokenizer_finished()) {
01008         // ... nein, Fehlerbehandlung
01009         tokenizer_error_print(current_linenum, NOT_ENOUGH_DATA);

01013         data_ptr.current = tokenizer_get_position();

01015     }
01016     // Tokenizer-Position wieder zuruecksetzen
