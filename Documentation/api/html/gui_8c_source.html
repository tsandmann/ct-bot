00178 static uint16_t gui_keypad_check(uint16_t rc5) {
00179     static uint8_t pressed = 0;
00180     static uint16_t last_rc5 = 0;
00181 
00182     if (keypad_callback == NULL) {
00183         /* keine Keypad-Eingabe aktiv */
00184         return rc5;
00185     }
00186 
00187     switch (rc5) {
00188     case RC5_CODE_DOT:
00189         keypad_mode = (uint8_t) ((keypad_mode + 1) & 1);
00190         return 0;
00191     case RC5_CODE_STOP:
00192         /* Abbruch */
00193         *keypad_buffer = 0;
00194         // no break
00195     case RC5_CODE_PLAY:
00196         /* fertig */
00197         keypad_callback(keypad_buffer);
00198         keypad_callback = NULL;
00199         memset(keypad_buffer, 0, 21);
00200         /* no break */
00201     case 0:
00202         /* keine neue Eingabe */
00203         return 0;
00204     }
00205 
00206     const uint8_t key = (uint8_t) (rc5 - RC5_CODE_0);
00207     if (key > 10) {
00208         /* ungueltige Eingabe */
00209         return 0;
00210     }
00211 
00212     if (!timer_ms_passed_32(&keypad_last_pressed, 1000) && rc5 == last_rc5) {
00213         /* dieselbe Taste wurde mehrfach gedrueckt */
00214         pressed++;
00215     } else {
00216         /* eins weiter schalten */
00217         keypad_result++;
00218         pressed = 0;
00219         LOG_DEBUG("neue Spalte:%u", keypad_result - keypad_buffer);
00220     }
00221 
00222     LOG_DEBUG("pressed=%u", pressed);
00223 
00224     if (pressed > 4) {
00225         /* Umlauf */
00226         pressed = 0;
00227     }
00228 
00229     LOG_DEBUG("key=%u", key);
00230     char data;
00231     while ((data = (char) ctbot_eeprom_read_byte(&gui_keypad_table[key][pressed])) == 0) {
00232         /* bei einigen Tasten ist nicht alles belegt */
00233         pressed = 0;
00234     }
00235 
00236     /* im numerischen Fall nur Ziffern auswaehlen */
00237     if (keypad_mode == 1) {
00238         uint8_t i = 4;
00239         while ((data = (char) ctbot_eeprom_read_byte(&gui_keypad_table[key][i])) == 0) {
00240             i--;
00241         }
00242         rc5 = 0; // naechsten Tastendruck als neue Taste registrieren
00243     }
00244 
00245     last_rc5 = rc5;
00246     keypad_last_pressed = TIMER_GET_TICKCOUNT_32;
00247     *keypad_result = data;
00248     LOG_DEBUG("data=%c", data);
00249     LOG_DEBUG("result=%s", keypad_buffer);
00250     return 0;
00251 }
