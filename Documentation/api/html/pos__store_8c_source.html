00060 pos_store_t * pos_store_create_size(Behaviour_t * owner, void * data, pos_store_size_t size) {
00061     if (owner == NULL) {
00062         LOG_ERROR("Fehler: owner==NULL!");
00063         return NULL;
00064     }
00065     if (size > POS_STORE_SIZE) {
00066         LOG_ERROR("Fehler: size > POS_STORE_SIZE");
00067         return NULL;
00068     }
00069     if (size & (size - 1)) {
00070         LOG_ERROR("Fehler: size keine 2er Potenz");
00071         return NULL;
00072     }
00073 
00074     LOG_DEBUG("Erzeuge Positionsspeicher (%u) fuer 0x%lx", size, (size_t) owner);
00075     pos_store_t * store = pos_store_from_beh(owner);
00076     if (store && store->mask != size - 1) {
00077         /* Positionsspeicher existiert bereits, aber mit anderer Groesse */
00078         pos_store_release(store);
00079         store = NULL;
00080     }
00081     if (store == NULL) {
00082         /* Verhalten hat noch keinen Positionsspeicher -> anlegen */
00083         pos_store_t * ptr;
00084         for (ptr=pos_stores; ptr<=&pos_stores[SLOT_COUNT-1]; ptr++) {
00085             if (ptr->owner == NULL) {
00086                 /* freien Platz gefunden */
00087                 store = ptr;
00088                 store->owner = owner;
00089                 store->mask = (pos_store_size_t) (size - 1);
00090                 LOG_DEBUG("verwende Slot %u", (store - pos_stores));
00091                 if (data == NULL) {
00092                     store->data = malloc(size * sizeof(position_t));
00093                     store->stat_data = 0;
00094                     LOG_DEBUG("verwende Heap-Speicher @ 0x%lx", (size_t) store->data);
00095                 } else {
00096                     store->data = data;
00097                     store->stat_data = 1;
00098                     LOG_DEBUG("verwende statischen Speicher @ 0x%lx", (size_t) data);
00099                 }
00100                 if (store->data == NULL) {
00101                     LOG_ERROR("Kein Speicher zur Verfuegung, Abbruch!");
00102                     pos_store_release(store);
00103                     return NULL;
00104                 }
00105                 break;
00106             }
00107         }
00108     }
00109 
00110     pos_store_clear(store);
00111     LOG_DEBUG("Positionsspeicher @ 0x%lx angelegt", (size_t) store);
00112     return store;
00113 }
