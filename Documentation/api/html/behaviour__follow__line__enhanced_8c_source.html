00058 static uint8_t check_line_sensors(void) {
00059     if (sensLineL >= LINE_SENSE || sensLineR >= LINE_SENSE) {
00060         line_state_enh = next_linestate; // jeweils naechster Eintrittspunkt
00061         return True;
00062     }
00063     return False;

00072 static uint8_t check_left_line_sensor(void) {
00073     if (sensLineL >= LINE_SENSE && sensLineR < LINE_SENSE)
00074         return True;
00075     return False;

00083 static uint8_t check_sensors(void) {
00084     // Unterbrechung wird daran erkannt, wenn sich der Linienfolger wegen keiner Linie mehr voraus nach links dreht und dabei
00085     // der rechte Abgrundsensor die unterbrochene Linie erkennt; Unterbrechung darf auch nur ca. maximal 3cm sein
00086     if (sensBorderR >= BORDER_DANGEROUS && sensLineL < LINE_SENSE && sensLineR
00087             < LINE_SENSE) {
00088         line_state_enh = CHECK_BREAK_IN_LINE;
00089         return True;
00090     }
00091 
00092     // Abstand seit letztem Check ermitteln; nur bei Ueberschreitung des Mindestabstandes seit letzter Hindernisumrundung
00093     // darf wegen Hindernis abgebrochen werden, da sonst nach Umrundung und gleichem Hindernis beim Ausrichten des Linienfolgers
00094     // dieser sonst gleich wieder moeglicherweise abgebrochen wird
00095     uint16_t diff = (uint16_t) get_dist(x_pos, y_pos, pos_on_line.x, pos_on_line.y);
00096     if ((diff >= 100 * 100) || (pos_on_line.x == 0 && pos_on_line.y == 0)) {
00097         // Merkpos auf 0, wenn Abstand ueberschritten wurde
00098         pos_on_line.x = 0;
00099         pos_on_line.y = 0;
00100         if (sensDistL <= OPTIMAL_DISTANCE || sensDistR <= OPTIMAL_DISTANCE) {
00101             line_state_enh = TURN_AROUND_HAZARD;
00102             return True;
00103         }
00104     }
00105 
00106     return False;
00107 }

00115 void bot_follow_line_enh_behaviour(Behaviour_t * data) {
00116     position_t point; // Var fuer Positionsberechnung nach Unterbrechung
00117 
00118     switch (line_state_enh) {
00119     case START_ENH: // maximal 60 cm vorwaerts bis Linie voraus
00120         bot_goto_dist(data, 600, 1);
00121         next_linestate = START_LINE_ENH; // Eintritt nachdem Cancel Linie gefunden hat
00122         bot_cancel_behaviour(data, bot_goto_pos_behaviour, check_line_sensors); // Ende der Vorwaertsfahrt wenn Linie erkannt wurde
00123         line_state_enh = END_LINE_ENH; // Ende falls keine Linie voraus gefunden wird
00124         break;
00125 
00126     case START_LINE_ENH: // Einsprung falls sich bot nun auf Linie befindet, weiter mit Linienfolger
00127         bot_follow_line(data); // Linienfolger laeuft los
00128         bot_cancel_behaviour(data, bot_follow_line_behaviour, check_sensors); // Abbruch Linienfolger bei Unterbrechung oder Hindernis
00129         line_state_enh = END_LINE_ENH; // einfach mal auf Ende, bei Cancel Linienfolger wird dort Einsprung gesetzt
00130         break;
00131 
00132     case CHECK_BREAK_IN_LINE: // Einsprung falls Linienunterbrechung erkannt wurde
00133         // Linie geht ja weiter nach Unterbrechung genau dort, wo rechter Abgrundsensor sich befindet; dort also Punkt berechnen und anfahren
00134         point = calc_point_in_distance(heading, BORDERSENSOR_POS_FW - 10,
00135                 -BORDERSENSOR_POS_SW);
00136         bot_goto_pos(data, point.x, point.y, 999);
00137         line_state_enh = START_LINE_ENH; // Linienfolger sollte hier wieder greifen nach Ueberwindung der Unterbrechung
00138         break;
00139 
00140     case TURN_AROUND_HAZARD: // auf jeden Fall erst mal in Ausgangsrichtung drehen
00144         bot_turn(data, 70); // links drehen
00145         line_state_enh = GO_AROUND_HAZARD; // naechsten Einsprung setzen
00146         break;
00147 
00148     case GO_AROUND_HAZARD: // etwas vorwaerts fahren um von Linie wegzukommen
00151         bot_goto_dist(data, 50, 1); // etwas vorwaerts fahren um von Linie wegzukommen
00152         line_state_enh = START_SOLVE_MAZE; // naechster Einsprung: Start solve_maze zum Umrunden des Hindernisses
00153         break;
00154 
00155     case START_SOLVE_MAZE: // Start Verhalten solve_maze zum Umrunden des Hindernisses
00156         bot_solve_maze(data); // Start solve_maze
00157         next_linestate = OTHER_SIDE_ON_LINE; // Eintritt nachdem Cancel Linie erkannt hat
00158         // Solve_maze abbrechen, wenn wieder Linie hinter Hindernis erkannt wird
00159         bot_cancel_behaviour(data, bot_solve_maze_behaviour, check_line_sensors);
00160         break;
00161 
00162     case OTHER_SIDE_ON_LINE: // Bot steht auf anderer Seite des Hindernisses wieder auf der Linie
00163         // hier muss der bot in die vom Hindernis wegweisende Richtung ausgerichtet werden
00164         // da er links um den Stein gegangen ist, blickt er entweder etwas Richtung Stein oder fast parallel;
00165         // einfach links wegdrehen, um ungefaehr die richtige Richtung einzunehmen
00166 
00167         // Sicherheit einbauen, dass er beim sich auf Linie Ausrichten und Hindernis sehen nicht wieder cancelt
00168         // Pos merken und erst wieder nach gewissem Abstand Hinderniserkennung mit Abbruch Linienfolger erlauben
00169         pos_on_line.x = x_pos;
00170         pos_on_line.y = y_pos;
00171         line_state_enh = TURN_NEXT_PART; // naechster Einsprung weiterdrehen mit Cancelueberwachung
00172         bot_turn(data, 40); // etwas links drehen
00173         break;
00174 
00175     case TURN_NEXT_PART: // weitere Drehung in richtiger Richtung fuer Linienfolger mit Cancelueberwachung
00176         // zweiter Teil der Drehung, hier aber Abbruch wenn linker Sensor auf Linie, da dies dann optimal fuer den Linienfolger ist
00177         bot_turn(data, 20);
00178         bot_cancel_behaviour(data, bot_turn_behaviour, check_left_line_sensor); // Abbruch Drehung wenn optimal fuer Linienfolger
00179         line_state_enh = SEARCH_LINE; // naechster Eintrittspunkt
00180         break;
00181 
00182     case SEARCH_LINE: // etwas vorfahren, um optimale Position fuer Linienfolger zu erreichen, falls er noch nicht optimal steht
00183         if (!check_left_line_sensor()) {
00184             bot_goto_dist(data, 50, 1); // etwas vor bis nur links Linie erkannt wird, dann optimal fuer Linienfolger
00185             bot_cancel_behaviour(data, bot_goto_pos_behaviour,
00186                     check_left_line_sensor); // Vorfahren abbrechen wenn optimal fuer Linienfolger
00187         }
00188         // eigentlich wurde hier alles getan, um den bot optimal fuer den Linienfolger hinzusetezn in die richtige Richtung nach Hindernisumrundung; mit dieser
00189         // Hoffnung gehts weiter mit dem Linienfolger
00190         line_state_enh = START_LINE_ENH; // naechster Einsprung->Linienfolger geht von vorn los
00191         break;
00192 
00193     default:
00194         return_from_behaviour(data);
00195         break;
00196     }
00197 
00198 }
