00151 static uint32_t check_fragmentation(uint32_t block, void * buffer) {
00152     LOG_DEBUG("block=0x%04x", block - 1);
00153     /* MBR lesen */
00154     mmc_read_sector(0, buffer);
00155     botfs_mbr_t * p_mbr = buffer;
00156     uint16_t first_sect = (uint16_t) p_mbr->part0.first_sect_offset;
00157     LOG_DEBUG("first_sect=0x%04x", first_sect);
00158 
00159     /* Bootsektor von Partition 1 lesen */
00160     mmc_read_sector(first_sect, buffer);
00161     botfs_fat16_bootsector_t * p_bs = buffer;
00162     p_bs->fat_name[7] = 0;
00163     if (strcmp_P(p_bs->fat_name, PSTR("FAT16  ")) != 0) {
00164 #ifdef DISPLAY_MINIFAT_INFO
00165         display_cursor(2, 1);
00166         display_puts("Kein FAT16");
00167 #endif
00168         LOG_ERROR("Keine FAT16-Partition");
00169 #ifdef LED_AVAILABLE
00170         LED_on(LED_TUERKIS);
00171 #endif  // LED_AVAILABLE
00172         return 0xffffffff;
00173     }
00174     uint16_t fat_offset = p_bs->reserved_sect + first_sect;
00175     LOG_DEBUG("fat_offset=0x%04x", fat_offset);
00176     uint16_t root_offset = fat_offset + p_bs->fat_copies * p_bs->sect_per_fat;
00177     LOG_DEBUG("root_offset=0x%04x", root_offset);
00178     uint16_t data_offset = root_offset + (p_bs->root_dir_entries * sizeof(botfs_fat16_dir_entry_t) + 511) / 512;
00179     LOG_DEBUG("data_offset=0x%04x", data_offset);
00180     uint8_t sect_per_cluster = p_bs->sect_per_cluster;
00181     LOG_DEBUG("sect_per_cluster=%u", sect_per_cluster);
00182 
00183     /* 1. Cluster der Datei berechnen */
00184     uint16_t first_cluster = (uint16_t) ((block - 1 - data_offset) / sect_per_cluster + 2);
00185     LOG_DEBUG("first_cluster=0x%04x", first_cluster);
00186 
00187     /* 1. FAT-Eintrag der Datei einlesen */
00188     uint16_t fat_block = fat_offset + first_cluster / (512 / sizeof(uint16_t));
00189     LOG_DEBUG("fat_block=0x%04x", fat_block);
00190     mmc_read_sector(fat_block, buffer);
00191     uint16_t entry_offset = first_cluster % (512 / sizeof(uint16_t));
00192     LOG_DEBUG("entry_offset=0x%04x", entry_offset);
00193     uint16_t * ptr = buffer;
00194     uint16_t entry = ptr[entry_offset];
00195 
00196     mmc_read_sector(block - 1, buffer); // Dateiheader wieder in Puffer laden
00197 
00198     LOG_DEBUG("Fat-Eintrag=0x%04x", entry);
00199     if (entry < 0xfff8 && entry != first_cluster + 1) {
00200         /* Datei ist fragmentiert */
00201 #ifdef DISPLAY_MINIFAT_INFO
00202         display_cursor(2, 1);
00203         display_puts("Datei fragmentiert  ");
00204 #endif
00205         LOG_ERROR("Datei ist fragmentiert!");
00206 #ifdef LED_AVAILABLE
00207         LED_on(LED_TUERKIS);
00208 #endif  // LED_AVAILABLE
00209         return 0xffffffff;
00210     }
00213     /* alles gut */
00214 #ifdef DISPLAY_MINIFAT_INFO
00215         display_cursor(2, 1);

00217 #endif
00218     LOG_DEBUG("Datei ist nicht fragmentiert");
