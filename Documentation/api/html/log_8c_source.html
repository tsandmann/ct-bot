00201 static void get_str_from_flash(const char * flash, char * ram, uint8_t n) {
00202     uint8_t i;
00203     /* Zeichen des Strings wortweise aus dem Flash holen */
00204     uint16_t * p_ram  = (uint16_t *) ram;
00205     uint16_t * p_flash = (uint16_t *) flash;
00206     for (i=0; i<n; i++){
00207         uint16_t tmp = pgm_read_word(p_flash++);
00208         *(p_ram++) = tmp;
00209         if ((uint8_t) tmp == 0 || (tmp & 0xFF00) == 0) break; // Stringende erreicht
00210     }
00211     *((char *) p_ram) = 0; // evtl. haben wir ein Byte zu viel gelesen, das korrigieren wir hier
