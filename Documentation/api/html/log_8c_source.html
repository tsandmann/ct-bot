00199 static void get_str_from_flash(const char * flash, char * ram, uint8_t n) {
00200     uint8_t i;
00201     /* Zeichen des Strings wortweise aus dem Flash holen */
00202     uint16_t * p_ram  = (uint16_t *) ram;
00203     uint16_t * p_flash = (uint16_t *) flash;
00204     for (i=0; i<n; i++){
00205         uint16_t tmp = pgm_read_word(p_flash++);
00206         *(p_ram++) = tmp;
00207         if ((uint8_t)tmp == 0 || (tmp & 0xFF00) == 0) break;    // Stringende erreicht
00208     }
00209     *((char *)p_ram) = 0;   // evtl. haben wir ein Byte zu viel gelesen, das korrigieren wir hier
