00098 void speed_control(uint8_t dev, int16_t * actVar, uint16_t * encTime, uint8_t i_time, uint8_t enc) {

00100     static int16_t lastErr[2] = {0,0};
00101     static int16_t last2Err[2] = {0,0};
00102     static uint8_t orignalTargetRate[2] = {0,0};
00103 
00104     if (encoderTargetRate[dev] == 0) {
00105         /* Fuer Stopp einfach alles nullsetzen */
00106         *actVar = 0;
00107         lastErr[dev] = 0;
00108         last2Err[dev] = 0;
00109     } else {
00110         /* Zeitdifferenz zwischen aktueller und ([vor- | 4.-])letzter Encoderflanke berechnen [176 us] */
00111         register uint8_t * p_time = (uint8_t *) encTime;
00112         uint16_t ticks_to_speed;
00113         uint16_t dt = *(uint16_t *) (p_time + i_time);  // aktueller Timestamp
00114         int8_t enc_correct = 0;
00115 
00116         /* Beim ersten Aufruf mit neuem Speed beruecksichtigen wir die Beschleunigung */
00117         if (start_signal[dev] == PID_START_DELAY) {
00118             orignalTargetRate[dev] = encoderTargetRate[dev];    // Zielgeschwindigkeit merken
00119             encoderTargetRate[dev] = BOT_SPEED_SLOW / 2;    // laaaangsam anfahren
00120 
00121 #ifdef SPEED_LOG_AVAILABLE
00122             /* Daten loggen */
00123             register uint8_t index = slog_i[dev];
00124             if (index < 24) {
00125                 slog->data[dev][index].encRate = 1; // Regelgroesse
00126                 slog->data[dev][index].err = encoderTargetRate[dev]; // Regeldifferenz
00127                 slog->data[dev][index].pwm = *actVar; // Stellgroesse
00128                 slog->data[dev][index].targetRate = encoderTargetRate[dev]; // Fuehrungsgroesse
00129                 slog->data[dev][index++].time = tickCount.u32; // Timestamp
00130                 slog_i[dev] = (uint8_t) (index > 24 ? 0 : index); // Z/25Z
00131             }
00132 #endif // SPEED_LOG_AVAILABLE
00133 
00134         } else { // 1. Aufruf => es gibt noch keinen korrekten Timestamp in der Vergangenheit => bis zum 2. Aufruf nix tun
00135             if (encoderTargetRate[dev] >= PID_SPEED_THRESHOLD) {
00136                 i_time = (uint8_t) (i_time - 4 * sizeof(encTime[0])); // Index 4.letzter Timestamp
00137                 ticks_to_speed = TICKS_TO_SPEED_2;
00138             } else if (encoderTargetRate[dev] < PID_SPEED_THRESHOLD / 2) {
00139                 i_time = (uint8_t) (i_time - sizeof(encTime[0])); // Index letzter Timestamp
00140                 ticks_to_speed = TICKS_TO_SPEED_0;
00141                 /* Regelgroesse korrigieren, wenn mit jeder Encoderflanke gerechnet wurde */
00142                 enc_correct = dev == 0 ? ENC_CORRECT_L : ENC_CORRECT_R;
00143                 if (enc == 1)  {
00144                     enc_correct = (int8_t) -enc_correct;
00145                 }
00146             } else {
00147                 i_time = (uint8_t) (i_time - 2 * sizeof(encTime[0])); // Index vorletzter Timestamp
00148                 ticks_to_speed = TICKS_TO_SPEED_1;
00149             }
00150             i_time = (uint8_t) (i_time & 0xf);                      // Index auf 4 Bit begrenzen
00151             dt -=  *(uint16_t *) (p_time + i_time); // gewaehlten Timestamp subtrahieren
00152 
00153             /* Bei Fahrt Regelgroesse berechnen */
00154             uint8_t encoderRate = (uint8_t) (ticks_to_speed / dt); // <dt> = [37; 800] -> <encoderRate> = [229; 10]
00155             if (encoderRate > 6) {
00156                 encoderRate = (uint8_t) (encoderRate + enc_correct);
00157             }
00158             /* Regeldifferenz berechnen */
00159             int16_t err = (encoderTargetRate[dev] - encoderRate);
00160             int16_t diff;
00161 #ifdef ADJUST_PID_PARAMS
00162             /* PID-Koeffizienten berechnen, falls PID-Parameter variabel */
00163             int16_t q0 = Kp + Kd / PID_Ta;
00164             int16_t q1 = -Kp - 2 * Kd / PID_Ta + Ki * PID_Ta;
00165             int16_t q2 = Kd / PID_Ta;
00166 
00167         /* Stellgroesse mit PID-Reglergleichung berechnen */
00168             diff = (q0 * err + q1 * lastErr[dev] + q2 * last2Err[dev]) >> PID_SHIFT;
00169 #else
00170             diff = (Q0 * err + Q1 * lastErr[dev] + Q2 * last2Err[dev]) >> PID_SHIFT;
00171 #endif // ADJUST_PID_PARAMS
00172             *actVar += diff;
00173 
00174             /* berechnete Stellgroesse auf zulaessige Werte begrenzen */
00175             if (*actVar > PWMMAX) *actVar = PWMMAX;
00176             else if (*actVar < PWMMIN) *actVar = PWMMIN;
00177 
00178             /* PWM-Lookup updaten */
00179             if (diff != 0 && encoderTargetRate[dev] == BOT_SPEED_SLOW/2 && start_signal[dev] == 0) {
00180                 uint8_t lastErrors = (uint8_t) (abs(last2Err[dev]) + abs(lastErr[dev]) + abs(err));
00181                 uint8_t turn = 0;
00182                 if (direction.left != direction.right) turn = 2;
00183                 if (pwm_values[dev + turn].rating >= lastErrors) {
00184                     pwm_values[dev + turn].pwm = (uint8_t) (*actVar >> 1);
00185                     pwm_values[dev + turn].rating = lastErrors;
00186                 }
00187             }
00188 
00189             /* Regeldifferenzen speichern */
00190             last2Err[dev] = lastErr[dev];
00191             lastErr[dev]  = err;
00192 
00193             /* Regelgroesse speichern, falls Displayausgabe gewuenscht */
00194 #ifdef DISPLAY_REGELUNG_AVAILABLE
00195             encoderRateInfo[dev] = encoderRate;
00196 #endif
00197 
00198 #ifdef SPEED_LOG_AVAILABLE
00199             /* Daten loggen */
00200             register uint8_t index = slog_i[dev];
00201             if (index < 24) {
00202                 slog->data[dev][index].encRate = encoderRate; // Regelgroesse
00203                 slog->data[dev][index].err = err; // Regeldifferenz
00204                 slog->data[dev][index].pwm = *actVar; // Stellgroesse
00205                 slog->data[dev][index].targetRate = encoderTargetRate[dev]; // Fuehrungsgroesse
00206                 slog->data[dev][index++].time = tickCount.u32; // Timestamp
00207                 slog_i[dev] = (uint8_t) (index > 24 ? 0 : index); // Z/25Z
00208             }
00209 #endif // SPEED_LOG_AVAILABLE
00210         }
00211 
00212         if (start_signal[dev] > 0) {
00213             start_signal[dev]--;
00214             /* langsam beschleunigen - eigentlich muesste man das sinusartig tun, aber das ist zu aufwendig
00215              * Verbesserungsmoeglichkeit: Faktoren bei START_DELAY optimieren (=> Sinus) */
00216             if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.75f)) {
00217                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +1/4
00218             } else if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.5f)) {
00219                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +2/4
00220             } else if (start_signal[dev] == (uint8_t) (PID_START_DELAY * 0.25f)) {
00221                 encoderTargetRate[dev] = (uint8_t) (encoderTargetRate[dev] + ((orignalTargetRate[dev] - BOT_SPEED_SLOW / 2) >> 2)); // +3/4
00222             } else if (start_signal[dev] == 0) {
00223                 encoderTargetRate[dev] = orignalTargetRate[dev];
00224             }
00225         }
00226     }
00227     /* PWM-Wert aktualisieren */
00228     motor_update(dev);
00229 }
