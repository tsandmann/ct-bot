00156 static void measure_distance(void) {
00157     if (measure_count == 0) {
00158         distL = 0;
00159         distR = 0;
00160     }
00161     /* auf neue Messung pruefen */
00162     if (last_toggle != sensDistLToggle) {
00163         last_toggle = sensDistLToggle;
00164         measure_count++;
00165     } else {
00166         return;
00167     }
00168     distL += (int16_t) ((float) sensDistL / 2.0f - volt_offset);
00169     distR += (int16_t) ((float) sensDistR / 2.0f - volt_offset);
00170     /* acht Messungen abwarten */
00171     if (measure_count == 8) {
00172         distL >>= 3;
00173         distR >>= 3;
00174         if (distL > 255 || distR > 255) {
