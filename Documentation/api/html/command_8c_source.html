00101 int8_t command_read(void) {
00102     int8_t bytesRcvd;
00103     int8_t start = 0; // Start des Kommandos
00104     int8_t i;
00105     command_t * command; // Pointer zum Casten der empfangegen Daten
00106     uint8_t buffer[RCVBUFSIZE]; // Puffer
00107 #ifdef PC
00108 #if BYTE_ORDER == BIG_ENDIAN
00109     uint16_t store; // Puffer fuer die Endian-Konvertierung
00110 #endif
00111 #endif // PC
00112 
00113     uint16_t old_ticks; // alte Systemzeit
00114     buffer[0] = 0; // Sicherheitshalber mit sauberem Puffer anfangen
00115 
00116     /* Daten holen, maximal soviele, wie ein Kommando lang ist */
00117     bytesRcvd = (int8_t) low_read(buffer, sizeof(command_t));
00118 
00119 #ifdef DEBUG_COMMAND_NOISY
00120     LOG_DEBUG("%d read", bytesRcvd);
00121     LOG_DEBUG("%x %x %x", buffer[0], buffer[1], buffer[2]);
00122 #endif
00123     /* Suche nach dem Beginn des Frames */
00124     while ((start < bytesRcvd) && (buffer[start] != CMD_STARTCODE)) {
00125         LOG_DEBUG("falscher Startcode");
00126         start++;
00127     }
00128 
00129     /* Wenn keine STARTCODE gefunden ==> Daten verwerfen */
00130     if (buffer[start] != CMD_STARTCODE) {
00131         LOG_DEBUG("kein Startcode");
00132         return -1;
00133     }
00134 
00135 #ifdef DEBUG_COMMAND_NOISY
00136     LOG_DEBUG("Start @%d", start);
00137 #endif
00138 
00139     // haben wir noch genug Platz im Puffer, um das Packet ferig zu lesen?
00140     if ((RCVBUFSIZE - (uint8_t) (start)) < sizeof(command_t)) {
00141         LOG_DEBUG("not enough space");
00142         return -1; // nein? ==> verwerfen
00143     }
00144 
00145     i = (int8_t) ((int8_t) sizeof(command_t) - (bytesRcvd - start));
00146 
00147     if (i > 0) { // Fehlen noch Daten ?
00148         LOG_DEBUG("Start @ %d es fehlen %d Bytes ", start, i);
00149         old_ticks = TIMER_GET_TICKCOUNT_16; // Systemzeit erfassen
00150 
00151         /* So lange Daten lesen, bis das Packet vollstaendig ist, oder der Timeout zuschlaegt */
00152         while (i > 0) {
00153             if (timer_ms_passed_16(&old_ticks, COMMAND_TIMEOUT)) {
00154                 /* Timeout ueberschritten */
00155                 LOG_DEBUG("Timeout beim Nachlesen");
00156                 return -1; // ==> Abbruch
00157             }
00158             LOG_DEBUG("%d Bytes missing", i);
00159             i = (int8_t) low_read(buffer + bytesRcvd, (uint8_t) i);
00160             LOG_DEBUG("%d read", i);
00161             bytesRcvd = (int8_t) (bytesRcvd + i);
00162             i = (int8_t) ((int8_t) sizeof(command_t) - (bytesRcvd - start));
00163         }
00164     }
00165 
00166 #ifdef DEBUG_COMMAND_NOISY
00167     LOG_DEBUG("%d/%d read/start", bytesRcvd, start);
00168     LOG_DEBUG("%x %x %x", buffer[start], buffer[start+1], buffer[start+2]);
00169 #endif
00170 
00171     command = (command_t *) (buffer + start); // Cast in command_t
00172 
00173 #ifdef DEBUG_COMMAND_NOISY
00174     LOG_DEBUG("start: %c ", command->startCode);
00175     //  command_display(command);
00176 #endif
00177 
00178     // validate (Startcode ist bereits ok, sonst waeren wir nicht hier)
00179     if (command->CRC == CMD_STOPCODE) {
00180 #ifdef DEBUG_COMMAND_NOISY
00181         LOG_DEBUG("Command is valid");
00182 #endif
00183 
00184 #ifdef CHECK_CMD_ADDRESS
00185         /* Ist das Paket ueberhaupt fuer uns? */
00186         if ((command->to != CMD_BROADCAST)
00187                 && (command->to != get_bot_address())
00188                 && (command->request.command != CMD_WELCOME)) {
00189             LOG_DEBUG("Fehler: Paket To= %d statt %u", command->to, get_bot_address());
00190 #ifdef LOG_AVAILABLE
00191             command_display(command);
00192 #endif
00193             return -1;
00194         }
00195 #endif // CHECK_CMD_ADDRESS
00196 
00197         // Transfer
00198         memcpy(&received_command, buffer + start, sizeof(command_t));
00199 #ifdef PC
00200 #if BYTE_ORDER == BIG_ENDIAN
00201         /* Umwandeln der 16 bit Werte in Big Endian */
00202         store = received_command.data_l;
00203         received_command.data_l = store << 8;
00204         received_command.data_l |= (store >> 8) & 0xff;
00205 
00206         store = received_command.data_r;
00207         received_command.data_r = store << 8;
00208         received_command.data_r |= (store >> 8) & 0xff;
00209 #endif // BYTE_ORDER == BIG_ENDIAN
00210 #endif // PC
00211         return 0;
00212     } else { // Command not valid
00213         LOG_ERROR("Invalid Command:");
00214         LOG_DEBUG("%x %x %x", command->startCode, command->request.command, command->CRC);
00215         return -1;
00216     }
00217 }
