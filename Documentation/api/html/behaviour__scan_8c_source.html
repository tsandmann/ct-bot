00061     static int16_t last_border_x, last_border_y, last_border_head;

00066     /* Verhalten je nach Cache-Fuellstand */
00067     uint8_t cache_free = (uint8_t) (map_update_fifo.size - map_update_fifo.count);
00068     if (cache_free < SCAN_OTF_CACHE_LEVEL_THRESHOLD) {
00069         if (cache_free == 1) {
00070             /* Cache ganz voll */
00071             if (scan_otf_modes.data.map_mode &&
00072                     sensBorderL < BORDER_DANGEROUS && sensBorderR < BORDER_DANGEROUS) {
00073                 /* Stoppe den Bot, damit wir Zeit haben die Karte einzutragen
00074                  * aber nur, wenn kein Abgrund erkannt wurde */

00076 #ifdef DEBUG_SCAN_OTF
00077                 LOG_DEBUG("Map-Cache voll, halte Bot an");
00078 #endif
00079                 /* Halte alle Verhalten eine Weile an, weil sie ja sonst evtl. weiterfahren wuerden */
00080                 os_thread_sleep(SCAN_OTF_SLEEP_TIME);
00081             } else {
00082                 /* Cache voll, neuen Eintrag verwerfen */
00083 #ifdef DEBUG_SCAN_OTF
00084                 LOG_DEBUG("Map-Cache voll, verwerfe neuen Eintrag");
00085 #endif
00086             }
00087             return;
00088         }
00089         /* Cache sehr voll */
00090         if (v_enc_left == 0 && v_enc_right == 0) {
00091             /* Falls Bot gerade steht, dann kleine Pause */
00092             os_thread_sleep(SCAN_OTF_SLEEP_TIME);
00093             return;
00094         }
00095     }
00096 
00097     /* Cache updaten, falls sich der Bot weit genug bewegt hat. */
00098     index++;
00099     if (index == MAP_UPDATE_CACHE_SIZE) {
00100         index = 0;
00101     }
00102     map_cache_t * cache_tmp = &map_update_cache[index];
00103     cache_tmp->mode.raw = 0;
00104     cache_tmp->dataL = 0;
00105     cache_tmp->dataR = 0;
00106 #ifdef MEASURE_POSITION_ERRORS_AVAILABLE
00107     cache_tmp->loc_prob = (uint8_t) (pos_error_radius < MAP_MAX_ERROR_RADIUS ? 255 - (pos_error_radius * 256
00108             / MAP_MAX_ERROR_RADIUS) : 0);
00109 #endif // MEASURE_POSITION_ERRORS_AVAILABLE
00110 
00111     /*
00112      * STANDFLAECHE
00113      * Die Standflaeche tragen wir nur ein, wenn der Bot auch ein Stueck gefahren ist
00114      */
00115     if (scan_otf_modes.data.location) {
00116         // ermitteln, wie weit der Bot seit dem letzten Location-Update gefahren ist
00117         uint16_t diff = (uint16_t) get_dist(x_pos, y_pos, last_location_x, last_location_y);
00118         if (diff > (SCAN_OTF_RESOLUTION_DISTANCE_LOCATION * SCAN_OTF_RESOLUTION_DISTANCE_LOCATION)) {
00119             // ist er weiter als SCAN_ONTHEFLY_DIST_RESOLUTION gefahren ==> Standflaeche aktualisieren
00120             cache_tmp->mode.data.location = 1;
00121             // Letzte Location-Update-Position sichern
00122             last_location_x = x_pos;
00123             last_location_y = y_pos;
00124         }
00125     }
00126 
00127     /*
00128      * DISTANZSENSOREN
00129      * Die Distanzsensoren tragen wir beim Geradeausfahren selten ein,
00130      * da sie viele Map-zellen ueberstreichen und das Eintragen teuer ist
00131      * und sie auf der anderen Seite (beim Vorwaertsfahren) wenig neue Infos liefern
00132     */
00133     if (scan_otf_modes.data.distance) {
00134         // ermitteln, wie weit der Bot gedreht hat
00135         int16_t turned = turned_angle(last_dist_head);
00136         // ermitteln, wie weit der Bot seit dem letzten distance-update gefahren ist
00137         uint16_t diff = (uint16_t) get_dist(x_pos, y_pos, last_dist_x, last_dist_y);
00138         if ((turned > SCAN_OTF_RESOLUTION_ANGLE_DISTSENS) ||
00139             (diff > (SCAN_OTF_RESOLUTION_DISTANCE_DISTSENS * SCAN_OTF_RESOLUTION_DISTANCE_DISTSENS))) {
00140             // Hat sich der Bot mehr als SCAN_ONTHEFLY_ANGLE_RESOLUTION gedreht ==> Blickstrahlen aktualisieren
00141             cache_tmp->mode.data.distance = 1;
00142 
00143             cache_tmp->dataL = (uint8_t) (sensDistL / 5);
00144             cache_tmp->dataR = (uint8_t) (sensDistR / 5);
00145             // Letzte Distanz-Update-Position sichern
00146             last_dist_x = x_pos;
00147             last_dist_y = y_pos;
00148             last_dist_head = heading_int;
00149         }
00150     }
00151 
00152     /*
00153      * ABGRUNDSENSOREN
00154      * Wir werten diese nur aus, wenn der Bot entweder
00155      * SCAN_OTF_RESOLUTION_DISTANCE_BORDER mm gefahren ist oder
00156      * SCAN_OTF_RESOLUTION_ANGLE_BORDER Grad gedreht hat
00157      */
00158     if (scan_otf_modes.data.border) {
00159         // ermitteln, wie weit der Bot seit dem letzten border-update gefahren ist
00160         uint16_t diff = (uint16_t) get_dist(x_pos, y_pos, last_border_x, last_border_y);
00161         // ermitteln, wie weit der Bot gedreht hat
00162         int16_t turned = turned_angle(last_border_head);
00163         if ((
00164               (diff > (SCAN_OTF_RESOLUTION_DISTANCE_BORDER * SCAN_OTF_RESOLUTION_DISTANCE_BORDER)) ||
00165               (turned > SCAN_OTF_RESOLUTION_ANGLE_BORDER)
00166             ) && ((sensBorderL > BORDER_DANGEROUS) || (sensBorderR > BORDER_DANGEROUS))) {
00167                 cache_tmp->mode.data.border = 1;
00168                 cache_tmp->mode.data.distance = 0;
00169                 cache_tmp->dataL = (uint8_t) (sensBorderL > BORDER_DANGEROUS);
00170                 cache_tmp->dataR = (uint8_t) (sensBorderR > BORDER_DANGEROUS);
00171 
00172                 last_border_x = x_pos;
00173                 last_border_y = y_pos;
00174                 last_border_head = heading_int;
00175         }
00176     }
00177 
00178     // ist ein Update angesagt?
00179     if (cache_tmp->mode.data.distance || cache_tmp->mode.data.location || cache_tmp->mode.data.border) {
00180         cache_tmp->x_pos = x_pos;
00181         cache_tmp->y_pos = y_pos;
00182 
00183 #ifdef MAP_USE_TRIG_CACHE
00184         if (cache_tmp->mode.data.distance || cache_tmp->mode.data.border) {
00185             cache_tmp->sin = heading_sin;
00186             cache_tmp->cos = heading_cos;
00187         }
00188 #else
00189         cache_tmp->heading = heading_10_int;
00190 #endif // MAP_USE_TRIG_CACHE
00191 
00192 #ifdef DEBUG_SCAN_OTF
00193         LOG_DEBUG("neuer Eintrag: x=%d y=%d head=%f distance=%d loaction=%d border=%d", cache_tmp->x_pos, cache_tmp->y_pos, cache_tmp->heading / 10.0f, cache_tmp->mode.distance, cache_tmp->mode.location, cache_tmp->mode.border);
00194 #endif
