00101

00106     if (sensBorderL > BORDER_DANGEROUS && sensBorderR > BORDER_DANGEROUS) {

00125 void bot_follow_wall_behaviour(Behaviour_t * data) {
00126     int16_t sensor = 0; // Merker fuer Abstandssensor je nach Wand links oder rechts
00127 
00128     if (check_function && (*check_function)()) // wenn ueberhaupt definiert und Abbruchbedingung erfuellt
00129         state = CHECK_FOR_READY;
00130 
00131     switch (state) {
00132 
00133     case CHECK_FOR_BACK: // Check auf zu nah oder bei Abgrund mit Rueckwaertsfahren
00134         // Der Bot erkennt Hindernis
00135         if (border_follow_wall_fired
00136                 || is_obstacle_ahead(OPTIMAL_DISTANCE + ADJUST_DISTANCE)) { // aus bot_olympic
00137             /* Abhaengig von der Seite, auf der die Wand ist, Entfernung ermitteln fuer rueckwaerts */
00138             sensor = (is_obstacle_ahead(OPTIMAL_DISTANCE + ADJUST_DISTANCE) < 0) ? sensDistL
00139                 : sensDistR ;
00140 
00141             if (border_follow_wall_fired) { // gesetzt durch registriertes Abgrundverhalten
00142                 state = CHECK_WALL_TURN; // nach rueckwaerts immer zum drehen
00143                 border_follow_wall_fired = False;
00144                 lastmsTime = TIMER_GET_TICKCOUNT_16; // Zeitzaehler geht jetzt los, damit mindestens xxms gedreht wird
00145                 break;
00146             } // Notfall erkannt
00147 
00148         } // Hindernis oder Abgrund vorhanden
00149 
00150         /* kam ich von Drehung, dann auch wieder dorthin sonst fahren */
00151         state = (uint8_t) (laststate == CHECK_WALL_TURN ? CHECK_WALL_TURN : CHECK_WALL_GO);
00152 
00153         break;
00154 
00155     case CHECK_WALL_GO:
00156 
00157         // Der Bot erkennt Hindernis oder Abgrund und muss gecheckt werden
00158         if (is_obstacle_ahead(OPTIMAL_DISTANCE + ADJUST_DISTANCE)) { // innerhalb xx cm Hindernis
00159 
00160             /* naechsten Status setzen und aktuellen merken */
00161             state = CHECK_FOR_BACK;
00162             laststate = CHECK_WALL_TURN;
00163 
00164             // Seite des Hindernisses ermitteln; <0 links, >= 0 rechts
00165             wall_side = is_obstacle_ahead(OPTIMAL_DISTANCE + ADJUST_DISTANCE); // Seite ermitteln
00166 
00167             /* bei Wand 90Grad voraus, dann per Zufall nach links oder rechts wegdrehen */
00168             if (sensDistL < COL_FAR && sensDistR < COL_FAR) {
00169                 wall_side = get_randomside(127); // < 127 links sonst rechts
00170             }
00171 
00172             break;
00173         }
00174 
00175         // Geschwindigkeiten je nach gesehener Entfernung setzen
00176         if (sensDistL > COL_FAR && sensDistR > COL_FAR) {
00177             speedWishLeft = BOT_SPEED_FAST;
00178             speedWishRight = BOT_SPEED_FAST;
00179         } else {
00180             if (sensDistL > COL_CLOSEST && sensDistR > COL_CLOSEST) {
00181                 speedWishLeft = BOT_SPEED_NORMAL;
00182                 speedWishRight = BOT_SPEED_NORMAL;
00183             } else {
00184                 speedWishLeft = BOT_SPEED_FOLLOW;
00185                 speedWishRight = BOT_SPEED_FOLLOW;
00186             }
00187         }
00188 
00189         laststate = CHECK_WALL_GO; // merken dass ich aus GO komme
00190 
00191         break; // immer weiter fahren in GO
00192 
00193     case CHECK_WALL_TURN:
00194         // nun bestimmte Zeit von Wand wegdrehen, je nach bereits ermittelter Seite
00195 
00196         // Setzen der Drehgeschwindigkeiten
00197         if (wall_side < 0) { // rechtsdrehen wenn Wand links
00198             speedWishLeft = BOT_SPEED_FOLLOW;
00199             speedWishRight = -BOT_SPEED_FOLLOW;
00200         } else { // sonst linksdrehen, wenn Wand rechts
00201             speedWishLeft = -BOT_SPEED_FOLLOW;
00202             speedWishRight = BOT_SPEED_FOLLOW;
00203         }
00204 
00205         // solange drehen bis Zeit in ms ueberschritten wird; ist dann genug Abstand bei beiden Sensoren
00206         // zur Wand vorhanden ist Drehung beendet
00207         // Durch die Zeit kommt der bot den Wandecken weniger nah, da er sich dadurch nicht genau parallel ausrichtet
00208         // und mit Drehen aufhoert wenn er ueber die Ecke schielt sondern sich wieder etwas von wegdreht; bei Abgrund dreht er
00209         // sich damit auch sicher weg mit hoeherer Drehzeit
00210         // Wartezeit hier nicht init. da dies dann noch zufaelliger wird
00211         if (timer_ms_passed_16(&lastmsTime, delay_time_turn)) {
00212             delay_time_turn = (uint16_t) (DELAY_NORMAL + TIMER_GET_TICKCOUNT_8); // wieder mit normalen Pseudo-Zufallswert belegen
00213 
00214             if (sensDistL > COL_NEAR&& sensDistR > COL_NEAR) { // keine Wand mehr in Sichtweite
00215                 state = CHECK_FOR_BACK; // Abgrund-/ Abstandscheck und dann los
00216                 laststate = 0; // Initialwert
00217                 break;
00218             }
00219         } // Check auf Ablauf der Wartezeit
00220 
00221         break;
00222 
00223     default:
00224         state = CHECK_FOR_BACK;
00225         laststate = 0;
00226         return_from_behaviour(data);
00227         break;
00228     }
00229 }
