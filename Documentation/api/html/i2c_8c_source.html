00051 ISR(TWI_vect) {
00052     uint8_t state = TWSR;
00053 #if I2C_PRESCALER != 0
00054     state &= 0xf8;
00055 #endif
00056     switch (state) {
00057         /* Start gesendet */
00058         case TW_START:
00059             /* no break */
00060         case TW_REP_START: {
00061             if (txSize > 0) {
00062                 /* Adresse+W senden */
00063                 TWDR = (uint8_t) (sl_addr & 0xfe); // W
00064             } else {
00065                 /* Adresse+R senden */
00066                 TWDR = sl_addr | 0x1;   // R
00067             }
00068             TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE);
00069             break;
00070         }
00071         /* Datum versandt, ACK empfangen */
00072         case TW_MT_DATA_ACK: {
00073             if (txSize == 0) {
00074                 if (rxSize == 0) {
00075                     /* Stopp Senden und beenden */
00076                     TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO); // Stopp senden
00077                     i2c_complete = 128; // Lock freigeben
00078                     break;
00079                 }
00080                 /* ReStart senden */
00081                 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE) | (1<<TWSTA); // ReStart senden
00082                 break;
00083             }
00084             /* no break */
00085         }
00086         /* SLA+W, ACK empfangen */
00087         case TW_MT_SLA_ACK: {
00088             /* Daten senden */
00089             TWDR = *pTxData;
00090             pTxData++;
00091             txSize--;
00092             TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE);
00093             break;
00094         }
00095         /* SLA+R, ACK empfangen */
00096         case TW_MR_SLA_ACK: {
00097             /* Auf Daten warten */
00098             if (rxSize > 1) {
00099                 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE) | (1<<TWEA);  // ACK senden
00100             } else {
00101                 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE);  // NACK senden
00102             }
00103             break;
00104         }
00105         /* Datum empfangen, ACK gesendet */
00106         case TW_MR_DATA_ACK: {
00107             /* Daten speichern */
00108             *pRxData = TWDR;
00109             pRxData++;
00110             rxSize--;
00111             if (rxSize > 0) {
00112                 /* es folgen noch weitere Daten, ACK senden */
00113                 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE) | (1<<TWEA);  // ACK senden
00114             } else {
00115                 /* das letzte Byte ist schon unterwegs */
00116                 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWIE);  // NACK senden
00117             }
00118             break;
00119         }
00120         /* Datum empfangen, NACK gesendet */
00121         case TW_MR_DATA_NACK: {
00122             /* Letztes Datum speichern */
00123             *pRxData = TWDR;
00124             TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO); // Stopp-Code senden
00125             i2c_complete = 128; // Lock freigeben
00126             break;
00127         }
00128         /* Fehler */
00129         default: {
00130             /* Abbruch */
00131             i2c_error = state;
00132             i2c_complete = 128; // Lock freigeben
00133             TWCR = (1<<TWINT);  // Int zuruecksetzen und I2C aus
00134         }
