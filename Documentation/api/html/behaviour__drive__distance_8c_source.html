00063 void bot_drive_distance_behaviour(Behaviour_t * data) {
00064     int16_t * encoder;
00065     int16_t to_drive;
00066 
00067     if (drive_distance_curve > 0) {
00068         // Es handelt sich um eine Rechtskurve, daher wird mit dem linken Encoder gerechnet
00069         encoder = &sensEncL;
00070     } else {
00071         encoder = &sensEncR;
00072     }
00073 
00074     to_drive = drive_distance_target - *encoder;
00075     if (drive_distance_speed < 0)
00076         to_drive = -to_drive;
00077 
00078     if (to_drive <= 0) {
00079         return_from_behaviour(data);
00080     } else {
00081         if ((drive_distance_speed > BOT_SPEED_SLOW || drive_distance_speed
00082                 < -BOT_SPEED_SLOW) && to_drive < (0.1 * ENCODER_MARKS))
00083             bot_drive(drive_distance_curve, drive_distance_speed / 2);
00084         else
00085             bot_drive(drive_distance_curve, drive_distance_speed);
