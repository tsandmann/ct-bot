00334 static void insert_behaviour_to_list(Behaviour_t * * list, Behaviour_t * behave) {
00335     Behaviour_t * ptr   = * list;
00336     Behaviour_t * temp;
00337 
00338     /* Kein Eintrag dabei? */
00339     if (behave == NULL) {
00340         return;
00341     }
00342 
00343     /* Erster Eintrag in der Liste? */
00344     if (ptr == NULL) {
00345         ptr = behave;
00346         *list = ptr;
00347     } else {
00348         /* Gleich mit erstem Eintrag tauschen? */
00349         if (ptr->priority < behave->priority) {
00350             behave->next = ptr;
00351             ptr = behave;
00352             *list = ptr;
00353         } else {
00354             /* Mit dem naechsten Eintrag vergleichen */
00355             while (NULL != ptr->next) {
00356                 if (ptr->next->priority < behave->priority) {
00357                     break;
00358                 }
00359 
00360                 /* Naechster Eintrag */
00361                 ptr = ptr->next;
00362             }
00363 
00364             temp = ptr->next;
00365             ptr->next = behave;
00366             behave->next = temp;
00367         }
00368     }

00442 static uint8_t isInCallHierarchy(Behaviour_t * job, Behaviour_t * caller_beh) {
00443     uint8_t level = 0;
00444 
00445     if (job == NULL) {
00446         LOG_DEBUG("kein Verhaltensdatensatz gegeben");
00447         return 0; // Liste ist leer
00448     }
00449 
00450     for (; job->caller; job = job->caller) {
00451         LOG_DEBUG("  ueberpruefe Verhalten %u mit caller %u...", job->priority, job->caller->priority);
00452         level++;
00453         if (job->caller == caller_beh) {
00454             LOG_DEBUG("   Verhalten %u wurde direkt von %u aufgerufen", job->priority, job->caller->priority);
00455             return level; // Direkter Aufrufer in Tiefe level gefunden
00456         }
00457     }
00458     return 0; // function kommt in Caller-Liste von job nicht vor
00459 } // O(n), n:=|Caller-Liste|

00466 void deactivate_called_behaviours(Behaviour_t * caller) {
00467     Behaviour_t * job; // Zeiger auf ein Verhalten
00468 
00469     LOG_DEBUG(""); // new line
00470     LOG_DEBUG("Callees von Verhalten %u sollen abgeschaltet werden.", get_behaviour(function)->priority);
00471     LOG_DEBUG("Beginne mit dem Durchsuchen der Liste");
00472     // Einmal durch die Liste gehen, und alle aktiven Funktionen pruefen, ob sie von dem uebergebenen Verhalten aktiviert wurden
00473     uint8_t i = 0;
00474     Behaviour_t * beh_of_function = NULL;
00475     for (job = behaviour; job; job = job->next) { // n mal
00476         if (job->active == BEHAVIOUR_ACTIVE) {
00477             i++;
00478             LOG_DEBUG("Verhalten mit Prio = %u ist ACTIVE, Durchlauf %u", job->priority, i);
00479             uint8_t level = isInCallHierarchy(job, caller); // O(n)
00480             LOG_DEBUG(" und hat Level %u Call-Abhaengigkeit", level);
00481             /* die komplette Caller-Liste (aber auch nur die) abschalten */
00482             Behaviour_t * ptr = job;
00483             for (; level > 0; --level) { // n mal
00484                 Behaviour_t * beh;
00485                 for (beh = behaviour; beh; beh = beh->next) { // n mal
00486                     /* Falls das Verhalten Caller eines anderen Verhaltens ist, duerfen wir es (noch) nicht deaktivieren! */
00487                     if (beh->caller == ptr) {
00488                         LOG_DEBUG("  Verhalten %u ist Caller eines anderen Verhaltens", ptr->priority);
00489                         break;
00490                     }
00491                 } // O(n)
00492                 Behaviour_t * tmp = ptr;
00493                 ptr = ptr->caller; // zur naechsten Ebene
00494                 if (beh == NULL) {
00495                     LOG_DEBUG("  Verhalten %u wird in Tiefe %u abgeschaltet", tmp->priority, level);
00496                     tmp->active = BEHAVIOUR_INACTIVE; // callee abschalten
00497                     tmp->caller = NULL; // Caller loeschen, damit Verhalten auch ohne BEHAVIOUR_OVERRIDE neu gestartet werden koennen
00498                 }
00499             }   // O(n^2)
00500             LOG_DEBUG(""); // new line
00501         }
00502         if (job == caller) {
00503             /* Verhalten fuer spaeter merken, wenn wir hier eh schon die ganze Liste absuchen */
00504             beh_of_function = job;
00505         }
00506     } // O(n^3)
00507     /* Verhaltenseintrag zu function benachrichtigen und wieder aktiv schalten */
00508     if (beh_of_function != NULL) {
00509         LOG_DEBUG("Verhalten %u wird aktiviert", beh_of_function->priority);
00510         beh_of_function->subResult = BEHAVIOUR_SUBCANCEL; // externer Abbruch
00511         beh_of_function->active = BEHAVIOUR_ACTIVE;
00512     }
00513 } // O(n^3)

00532 Behaviour_t * switch_to_behaviour(Behaviour_t * from, void (* to)(Behaviour_t *), uint8_t mode) {
00533     LOG_DEBUG("switch_to_behaviour(0x%lx, 0x%lx, %u)", (size_t) from, (size_t) to, mode);
00534     Behaviour_t * job = get_behaviour(to);
00535     if (job == NULL) {
00536         /* Zielverhalten existiert gar nicht */
00537         if (from) {
00538             from->subResult = BEHAVIOUR_SUBFAIL;
00539         }
00540         return NULL;
00541     }
00542 
00543     bit_t tmp;
00544     behaviour_mode_t beh_mode;
00545     tmp.byte = (uint8_t) (mode & 1);
00546     beh_mode.override = tmp.bit;
00547     tmp.byte = (uint8_t) ((mode & 2) >> 1);
00548     beh_mode.background = tmp.bit;
00549 
00550     if (job->caller) { // Ist das auzurufende Verhalten noch beschaeftigt?
00551         if (beh_mode.override == BEHAVIOUR_NOOVERRIDE) { // nicht ueberschreiben, sofortige Rueckkehr
00552             if (from) {
00553                 from->subResult = BEHAVIOUR_SUBFAIL;
