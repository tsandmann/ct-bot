00176 static inline uint32_t mmc_get_mmcblock_of_page(uint32_t addr) {
00177 #ifdef MCU
00178     /* Eine effizientere Variante von addr >> 9 */
00179     asm volatile(
00180         "mov %A0, %B0   \n\t"
00181         "mov %B0, %C0   \n\t"
00182         "mov %C0, %D0   \n\t"
00183         "lsr %C0        \n\t"
00184         "ror %B0        \n\t"
00185         "ror %A0        \n\t"
00186         "clr %D0            "
00187         :   "=r"    (addr)
00188         :   "0"     (addr)
00189     );
00190     return addr;
00191 #else
00192     return addr >> 9;
00193 #endif // MCU
00194 }

00325     uint32_t start_addr;
00326     if (/*aligned == 0 || */mmc_get_end_of_page(next_mmc_address) == mmc_get_end_of_page(next_mmc_address+size-1)) {
00327         /* Daten einfach an der naechsten freien virtuellen Adresse speichern */
00328         start_addr = next_mmc_address;
00329     } else {
00330         /* Rest der letzten Seite ueberspringen und Daten in neuem Block speichern */
00331         start_addr = mmc_get_end_of_page(next_mmc_address) + 1;
00332     }   
00333     if (start_addr+size > swap_space) return 0; // wir haben nicht mehr virtuellen Speicher als Platz auf dem Swap-Device
00334     /* interne Daten aktualisieren */
00335     next_mmc_address = start_addr + size;
00336 #ifdef VM_STATS_AVAILABLE
00337     stats_data.vm_used_bytes = next_mmc_address - mmc_start_address;
00338 #endif
00339     return start_addr;
