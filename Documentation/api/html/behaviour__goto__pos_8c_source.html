00087 void bot_goto_pos_behaviour(Behaviour_t * data) {
00088     static int32_t done;
00089     static int16_t v_m;
00090     static int16_t v_l;
00091     static int16_t v_r;
00092 
00093     /* Abstand zum Ziel berechnen (als Metrik euklidischen Abstand benutzen) */
00094     int32_t diff_to_target = get_dist(dest_x, dest_y, x_pos, y_pos);
00095     LOG_DEBUG("diff_to_target=%ld", diff_to_target);
00096 
00097     /* gefahrene Strecke berechnen */
00098     int32_t driven = last_diff_to_target - diff_to_target;
00099     LOG_DEBUG("last_diff_to_t=%ld", last_diff_to_target);
00100     LOG_DEBUG("driven=%ld", driven);
00101     last_diff_to_target = diff_to_target;
00102 
00103     /* Pruefen, ob wir schon am Ziel sind */
00104     if (state != END && diff_to_target <= (int32_t) margin) {
00105         state = LAST_TURN;
00106     }
00107 
00108     switch (state) {
00109     case FIRST_TURN: {
00110         /* ungefaehr in die Zielrichtung drehen */
00111         LOG_DEBUG("first turn");
00112         int16_t alpha = (int16_t) calc_angle_diff(dest_x - x_pos, dest_y - y_pos);
00113         if (drive_dir < 0) {
00114             /* Winkelkorrektur, falls rueckwaerts */
00115             alpha += 180;
00116             if (alpha > 180) {
00117                 alpha -= 360;
00118             }
00119         }
00120         LOG_DEBUG("alpha=%d", alpha);
00121         state = CALC_WAY;
00122         if (diff_to_target < straight_go) {
00123             LOG_DEBUG("bot_turn(%d)", alpha);
00124             bot_turn(data, alpha);
00125             return;
00126         }
00127         if (abs(alpha) > max_angle) {
00128             int16_t to_turn = abs(alpha) - (max_angle - 10);
00129             if (alpha < 0) {
00130                 to_turn = -to_turn;
00131             }
00132             bot_turn(data, to_turn);
00133             LOG_DEBUG("bot_turn(%d)", to_turn);
00134             break;
00135         }
00136     }
00137     /* no break */
00138     case CALC_WAY: {
00139         /* Kreisbogenfahrt zum Ziel berechnen */
00140         LOG_DEBUG("calc way...");
00141         /* Winkel- und Streckenbezeichnungen wie in -> Documentation/bot_goto_pos.png */
00142         int16_t diff_x = dest_x - x_pos;
00143         int16_t diff_y = dest_y - y_pos;
00144         float sin_alpha = heading_sin;
00145         float tan_alpha = heading_sin / heading_cos;
00146         if (drive_dir < 0) {
00147             sin_alpha = -sin_alpha;
00148             LOG_DEBUG("alpha=%f", heading + 180.0f);
00149         } else {
00150             LOG_DEBUG("alpha=%f", heading);
00151         }
00152         float beta = calc_angle_diff_rad(diff_x, diff_y); // [-M_PI; M_PI]
00153         if (drive_dir < 0) {
00154             LOG_DEBUG("beta=%f", deg(beta) + 180.0f);
00155         } else {
00156             LOG_DEBUG("beta=%f", deg(beta));
00157         }
00158         const float gamma = (float) M_PI_2 - beta - rad(heading);
00159         if (sin_alpha == 0.0f) {
00160             sin_alpha = 0.000001f;
00161         }
00162         /* gegenueber Documentation/bot_gotp_pos.png enthalten alle Variablen mit dem Suffix "_2" den doppelten Wert ("/2" ausgeklammert) */
00163         const int16_t h1_2 = diff_y;
00164         const int16_t h2_2 = diff_x;
00165         const float h4_2 = h1_2 * tan_alpha;
00166         const float h5_2 = h4_2 / sin_alpha;
00167         const float h6_2 = (h2_2 + h4_2) * sinf(gamma);
00168         float sin_beta = sinf(beta);
00169         if (sin_beta == 0.0f) {
00170             sin_beta = 0.000001f;
00171         }
00172         const float h7_2 = h6_2 / sin_beta;
00173         float radius = (h5_2 + h7_2) / 2.0f;
00174         LOG_DEBUG("radius=%f", radius);
00175         if ((int16_t) radius == 0) {
00176             radius = 100000.0f; // geradeaus
00177         }
00178         /* Geschwindigkeit an Entfernung zum Zielpunkt anpassen */
00179         float x = diff_to_target < 360L * 360L ? (float) diff_to_target / ((360.0f / M_PI * 2.0f) * (360.0f / M_PI
00180             * 2.0f)) : M_PI / 2; // (0; pi/2]
00181         v_m = (int16_t) (sinf(x) * (float) (v_m_max - v_m_min)); // [0; v_m_max - v_m_min]
00182         v_m += v_m_min; // [v_m_min; v_m_max]
00183         if (drive_dir < 0) {
00184             v_m = -v_m;
00185             radius = -radius;
00186         }
00187         /* Geschwindigkeiten auf die beiden Raeder verteilen, um den berechneten Radius der Kreisbahn zu erhalten */
00188         v_l = iroundf(radius / (radius + ((float) WHEEL_TO_WHEEL_DIAMETER / 2.0f)) * (float) v_m);
00189         v_r = iroundf(radius / (radius - ((float) WHEEL_TO_WHEEL_DIAMETER / 2.0f)) * (float) v_m);
00190 
00191         int16_t vl_abs = abs(v_l);
00192         int16_t vr_abs = abs(v_r);
00193         /* Wenn der Radius zu klein wird, bekommen wir fuer die Raeder Geschwindigkeiten,
00194          * die kleiner bzw. groesser als moeglich sind, auesserdem faehrt der Bot dann
00195          * eher Slalom, darum beginnen wir in diesem Fall neu mit dem Verhalten.
00196          */
00197         if (vl_abs < v_min || vr_abs < v_min || vl_abs > v_max || vr_abs > v_max) {
00198             state = FIRST_TURN;
00199             LOG_DEBUG("Geschwindigkeiten ungueltig, beginne neu");
00200             LOG_DEBUG("v_l=%d\tv_r=%d\tv_m=%d", v_l, v_r, v_m);
00201             return;
00202         }
00203         /* Statusupdate */
00204         done = 0;
00205         state = RUNNING;
00206     }
00207     /* no break */
00208     case RUNNING: {
00209         /* Berechnete Geschwindigkeiten setzen */
00210         LOG_DEBUG("v_l=%d; v_r=%d", v_l, v_r);
00211         LOG_DEBUG("x_pos=%d; y_pos=%d", x_pos, y_pos);
00212         speedWishLeft = v_l;
00213         speedWishRight = v_r;
00214         /* Alle sqrt(recalc_dist) mm rechnen wir neu, um Fehler zu korrigieren */
00215         done += labs(driven);
00216         if (done > recalc_dist) {
00217             state = CALC_WAY;
00218         }
00219         break;
00220     }
00221     case LAST_TURN: {
00222         /* Nachlauf abwarten */
00223         LOG_DEBUG("Nachlauf abwarten...");
00224         speedWishLeft = BOT_SPEED_STOP;
00225         speedWishRight = BOT_SPEED_STOP;
00226         if (v_enc_left != 0 || v_enc_right != 0) {
00227             return;
00228         }
00229 
00230         int16_t diff = (int16_t) diff_to_target;
00231         int16_t last_diff = (int16_t) last_diff_to_target;
00232         if (last_diff < (int16_t) driven || (int16_t) driven < 0) {
00233             /* zu weit gefahren */
00234             diff = -diff;
00235         }
00236         LOG_DEBUG("Fehler^2=%d mm^2", diff);
00237         /* Aus Fehler neuen Korrekturwert berechnen und im EEPROM speichern */
00238         const int16_t error = (int16_t) margin;
00239         LOG_DEBUG("error=%d", error);
00240         int16_t new_error = (error + diff) / 2;
00241         LOG_DEBUG("new_error=%d", new_error);
00242         if (new_error < MIN_TARGET_MARGIN) {
00243             new_error = MIN_TARGET_MARGIN;
00244         } else if (new_error > MAX_TARGET_MARGIN) {
00245             new_error = MAX_TARGET_MARGIN;
00246         }
00247         if (new_error != error) {
00248             ctbot_eeprom_update_word(p_goto_pos_err, (uint16_t) new_error);
00249             LOG_DEBUG("EEPROM: new_error=%d", new_error);
00250         }
00251         /* fast fertig, evtl. noch drehen */
00252         if (dest_head != 999) {
00253             /* Noch in die gewuenschte Blickrichtung drehen */
00254             int16_t to_turn = (int16_t) (dest_head - heading_int);
00255             if (to_turn > 180) {
00256                 to_turn = -360 + to_turn;
00257             } else if (to_turn < -180) {
00258                 to_turn += 360;
00259             }
00260             LOG_DEBUG("to_turn=%d", to_turn);
00261             bot_turn(data, to_turn);
00262         }
00263         state = END;
00264         break;
00265     }
00266     default:
00267         drive_dir = 1;
00268         return_from_behaviour(data);
00269         return;
00270     }
00271 }

00282 Behaviour_t * bot_goto_pos(Behaviour_t * caller, int16_t x, int16_t y, int16_t head) {
00283     dest_x = x;
00284     dest_y = y;
00285     dest_head = head;
00286 
00287     /* Verhalten starten */
00288     Behaviour_t * beh = switch_to_behaviour(caller, bot_goto_pos_behaviour, BEHAVIOUR_OVERRIDE);
00289 
00290     /* Inits */
00291     if (state != END) {
00292         drive_dir = 1;  // unsanfter Abbruch beim letzten Mal
00293         LOG_DEBUG("Richtung unbekannt, nehme vorwaerts an");
00294         LOG_DEBUG("state=%u", state);
00295     }
00296     state = FIRST_TURN;
00297 
00298     int32_t diff_to_target = get_dist(dest_x, dest_y, x_pos, y_pos);
00299     last_diff_to_target = diff_to_target;
00300     LOG_DEBUG("diff_to_target=%ld", diff_to_target);
00301     if (diff_to_target > straight_go) {
00302         /* fuer grosse Strecken zweiten Fehlerwert verwenden */
00303         p_goto_pos_err = &goto_pos_err[1];
00304     } else {
00305         p_goto_pos_err = &goto_pos_err[0];
00306     }
00307     margin = ctbot_eeprom_read_word(p_goto_pos_err);
00308     LOG_DEBUG("margin=%u", margin);
00309 
00310     LOG_DEBUG("(%d mm|%d mm|%d Grad)", x, y, head);
00311     if (drive_dir >= 0) {
00312         LOG_DEBUG("vorwaerts");
00313     } else {
00314         LOG_DEBUG("rueckwaerts");
00315     }
00316 
00317     return beh;

00342 Behaviour_t * bot_goto_dist(Behaviour_t * caller, int16_t distance, int8_t dir) {
00343     drive_dir = dir;
00344 
00345     /* Zielpunkt aus Blickrichtung und Distanz berechnen */
00346     int16_t target_x = (int16_t) (distance * heading_cos);
00347     int16_t target_y = (int16_t) (distance * heading_sin);
00348     if (drive_dir < 0) {
00349         // rueckwaerts
00350         target_x = -target_x;
00351         target_y = -target_y;
00352     }
00353     target_x += x_pos;
00354     target_y += y_pos;
00355     LOG_DEBUG("Zielpunkt=(%d|%d)", target_x, target_y);
00356     LOG_DEBUG("Richtung=%d", drive_dir);
00357     state = END;
00358     /* Verhalten starten */
00359     return bot_goto_pos(caller, target_x, target_y, heading_int);
