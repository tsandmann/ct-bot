00269 static void remotecall_convert_params(uint8_t * dest, const uint8_t count, const uint8_t * len, const void * data) {
00270     const uint8_t * ptr = data;
00271     (void) ptr;
00272     LOG_DEBUG("p_data=%x %x %x %x", ptr[0], ptr[1], ptr[2], ptr[3]);
00273     LOG_DEBUG("%x %x %x %x", ptr[4], ptr[5], ptr[6], ptr[7]);
00274     LOG_DEBUG("%x %x %x %x", ptr[8], ptr[9], ptr[10], ptr[11]);
00275 #ifdef MCU
00276     dest += 8; // ans Ende springen
00277     uint8_t i;
00278     /* Parameter rueckwaerts einlesen */
00279     for (i = 0; i < count; ++i) {
00280         uint8_t pos = (uint8_t) (len[i] > 1 ? len[i] : 2); // alle Parameter sind 16-Bit-aligned
00281         dest -= pos;
00282         memcpy(dest, ptr, len[i]);
00283         ptr += sizeof(remote_call_data_t);
00284     }
00285 #else // PC
00286     (void) len;
00287 #if BYTE_ORDER == LITTLE_ENDIAN
00288     /* Daten einfach kopieren */
00289     memcpy(dest, data, count * sizeof(remote_call_data_t));
00290 #else // BIG_EDIAN
00291     uint8_t i;
00292     for (i = 0; i < count; ++i) {
00293         /* Parameter i von little-endian nach big-endian konvertieren*/
00294         remote_call_data_t in, out;
00295         in = *((remote_call_data_t *) data);
00296         out.u32 = ((in.u32 & 0xff) << 24) | ((in.u32 & 0xff00) << 8) | ((in.u32 & 0xff0000) >> 8) | ((in.u32 & 0xff000000) >> 24);
00297         memcpy(dest, &out, sizeof(remote_call_data_t));
00298         dest += sizeof(remote_call_data_t);
00299         data += sizeof(remote_call_data_t);
00300     }
00301 #endif // LITTLE_EDIAN
00302 
00303 #endif // MCU
00304 }

00324 void bot_remotecall_behaviour(Behaviour_t * data) {
00325     LOG_DEBUG("Enter bot_remotecall_behaviour");
00326 
00327     switch (running_behaviour) {
00328         case REMOTE_CALL_SCHEDULED: // Es laueft kein Auftrag, aber es steht ein Neuer an
00329             LOG_DEBUG("REMOTE_CALL_SCHEDULED");

00332                 LOG_DEBUG("keine Funktion gefunden. Exit");
00333                 running_behaviour = REMOTE_CALL_IDLE;
00334                 exit_behaviour(data, BEHAVIOUR_SUBFAIL);

00344             // Auf dem MCU liegt die remotecall_beh_list-Struktur im Flash und muss erst geholt werden

00347 #endif // PC
00348 
00349             if (parameter_count > REMOTE_CALL_MAX_PARAM) {
00350                 LOG_DEBUG("Parameteranzahl unzulaessig!");
00351                 running_behaviour = REMOTE_CALL_IDLE;

00354 
00355             LOG_DEBUG("function_id=%u", function_id);

00361 #else // MCU

00363 #endif // PC
00364             running_behaviour = REMOTE_CALL_RUNNING;

00374             function_name = (char *) &tmp;
00375 
00376 #ifdef COMMAND_AVAILABLE
00377             int16_t result = data->subResult;

00379                 /* kein Caller, also kam der Aufruf wohl vom Sim */
00380                 command_write_data(CMD_REMOTE_CALL, SUB_REMOTE_CALL_DONE, result, result, function_name);
00381             }

00544 void remotecall_display(void) {
00545     static uint8_t first_row = 0; // erste Verhaltens-ID, die auf dem Display angezeigt wird
00546     static int8_t selected_row = 0; // Offset der markierten Zeile (relativ zu first_row)
00547 
00548     /* Keyhandler */
00549     switch (RC5_Code) {
00550     case RC5_CODE_UP: // zurueck blaettern
00551         --selected_row;
00552         keypad_param_index = 0;
00553         beh_selected = 0;
00554         RC5_Code = 0;
00555         break;
00556 
00557     case RC5_CODE_DOWN: // vor blaettern oder laufenden RC abbrechen
00558         if (running_behaviour == REMOTE_CALL_RUNNING) {
00559             bot_remotecall_cancel();
00560         } else {
00561             ++selected_row;
00562             keypad_param_index = 0;
00563             beh_selected = 0;
00564         }
00565         RC5_Code = 0;
00566         break;
00567 
00568     case RC5_CODE_PLAY: // auswahlen / starten
00569         beh_selected = 1;
00570         first_row = (uint8_t) (first_row + selected_row);
00571         if (first_row > STORED_CALLS - 2) {
00572             first_row = (uint8_t) (first_row - (STORED_CALLS - 1));
00573         }
00574         selected_row = 0;
00575         RC5_Code = 0;
00576         break;
00577     }
00578 
00579     /* Anzeige */
00580     display_clear();
00581     if (selected_row > 2) {
00582         selected_row = 2;
00583         ++first_row;
00584         if (first_row > STORED_CALLS - 2) {
00585             first_row = 0;
00586         }
00587     } else if (selected_row < 0) {
00588         selected_row = 0;
00589         --first_row;
00590         if (first_row > STORED_CALLS - 2) {
00591             first_row = STORED_CALLS - 2;
00592         }
00593     }
00594

00596     const remotecall_entry_t * ptr_call;
00597     const uint8_t n = (uint8_t) (beh_selected > 0 || running_behaviour == REMOTE_CALL_RUNNING ? 1 : 3);
00598     uint8_t i;
00599     for (i = 1; i <= n; ++i) {
00600         if (running_behaviour == REMOTE_CALL_RUNNING) {
00601             row = function_id;
00602         } else {
00603             row = (uint8_t) (first_row + (i - 1));
00604             if (row > STORED_CALLS - 2) {
00605                 row = (uint8_t) (row - (STORED_CALLS - 1));
00606             }
00607         }
