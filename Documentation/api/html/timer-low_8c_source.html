00107 void timer_2_init(void) {
00108     TCNT2 = 0x00; // TIMER vorladen
00109 
00110     // aendert man den Prescaler muss man die Formel fuer OCR2 anpassen !!!
00111     // Compare Register nur 8-Bit breit --> evtl. Teiler anpassen
00112 #if defined MCU_ATMEGA644X || defined __AVR_ATmega1284P__
00113     TCCR2A = _BV(WGM21);    // CTC Mode
00114     TCCR2B = _BV(CS22);     // Prescaler = CLK/64
00115     OCR2A = ((XTAL/64/TIMER_2_CLOCK) - 1);  // Timer2A
00116     TIMSK2  |= _BV(OCIE2A); // TIMER2 Output Compare Match A Interrupt an
00117 #else
00118     // use CLK/64 prescale value, clear timer/counter on compare match
00119     TCCR2 = _BV(WGM21) | _BV(CS22);
00120     OCR2 = ((XTAL/64/TIMER_2_CLOCK) - 1);
00121     TIMSK  |= _BV(OCIE2);   // enable Output Compare 0 overflow interrupt
00122 #endif // MCU_ATMEGA644X || ATmega1284P
00123 
00124     __builtin_avr_sei(); // enable interrupts
