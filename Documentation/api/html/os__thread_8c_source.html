00065 Tcb_t * os_create_thread(void * pStack, void (* pIp)(void)) {
00066     os_enterCS(); // Scheduler deaktivieren
00067     uint8_t i;
00068     Tcb_t * ptr = os_threads;
00069     // os_scheduling_allowed == 0 wegen os_enterCS()
00070     for (i = os_scheduling_allowed; i < OS_MAX_THREADS; ++i, ++ptr) {
00071         if (ptr->stack == NULL) {
00072             ptr->wait_for = &dummy_signal; // wait_for belegen
00073             if (os_thread_running == NULL) {
00074                 /* Main-Thread anlegen (laeuft bereits) */
00075                 os_thread_running = ptr;
00076                 ptr->stack = pStack;
00077 #ifdef OS_DEBUG
00078                 os_mask_stack(os_kernel_stack, OS_KERNEL_STACKSIZE);
00079 #endif
00080             } else {
00081                 /* "normalen" Thread anlegen */
00082                 os_ip_t tmp;
00083                 tmp.ip = pIp;
00084                 /* Stack so aufbauen, als waere der Thread bereits einmal unterbrochen worden */
00085                 uint8_t * sp = pStack; // Entry-Point des Threads
00086                 *sp = tmp.bytes.lo8; // Return-Adresse liegt in Big-Endian auf dem Stack!
00087                 *(sp - 1) = tmp.bytes.hi8;
00088                 tmp.ip = &os_exitCS; // setzt os_scheduling_allowed auf 1, nachdem der Thread das erste Mal geschedult wurde
00089                 *(sp - 2) = tmp.bytes.lo8;
00090                 *(sp - 3) = tmp.bytes.hi8;
00091                 *(sp - 4) = SREG; // beim ersten Wechsel auf diesen Thread wird das Statusregister vom Stack geholt, darum hier auf den Stack schreiben
00092                 uint8_t * p = pStack;
00093                 ptr->stack = p - (4 + OS_CONTEXT_SIZE); // 4x push und OS_CONTEXT_SIZE Bytes fuer Kontext => Stack-Pointer - (4 + OS_CONTEXT_SIZE)
00094             }
00095             os_scheduling_allowed = 1; // Scheduling wieder erlaubt
00096             /* TCB zurueckgeben */
00097             return ptr;
00098         }
00099     }
00100     os_scheduling_allowed = 1; // Scheduling wieder erlaubt
00101     return NULL; // Fehler :(
00102 }

00119 void os_thread_yield(void) {
00120     os_enterCS();
00121     uint32_t now = TIMER_GET_TICKCOUNT_32;
00122     /* Zeitpunkt der letzten Ausfuehrung nur in 16 Bit, da yield() nur Sinn macht, wenn es
00123      * regelmaessig aufgerufen wird, sonst sleep() benutzen. */
00124     uint16_t runtime = (uint16_t) now - (uint16_t) os_thread_running->lastSchedule;
00125     if (runtime > MS_TO_TICKS(OS_TIME_SLICE)) {
00126         /* Zeitscheibe wurde bereits ueberschritten ==> kein Threadwechsel */
00127 #ifdef OS_KERNEL_LOG_AVAILABLE
00128         LOG_DEBUG("%u missed dl\t%5uus", os_thread_running - os_threads, runtime * TIMER_STEPS);
00129 #endif // OS_KERNEL_LOG_AVAILABLE
00130 #ifdef MEASURE_UTILIZATION
00131         os_thread_running->statistics.runtime += runtime;
00132         /* Zaehler fuer verpasste Deadlines erhoehen */
00133         os_thread_running->statistics.missed_deadlines++;
00134 #endif // MEASURE_UTILIZATION
00135         /* Timestamp zuruecksetzen */
00136         os_thread_running->lastSchedule = (uint16_t) now;
00137     } else {
00138         /* Wir haben noch Zeit frei, die wir dem naechsten Thread schenken koennen */
00139         os_thread_running->nextSchedule = now + (uint16_t) (MS_TO_TICKS(OS_TIME_SLICE) - runtime);
00140         /* Scheduler wechselt die Threads, Aufruf am Ende der Funktion */
00141         os_scheduling_allowed = 2;
00142     }
00143     /* os_exitCS() */
00144     if (test_and_set((uint8_t *) &os_scheduling_allowed, 1) == 2) {
00145         os_schedule(now);
00146     }
00147 }

00166 void os_switch_thread(Tcb_t * from, Tcb_t * to) {
00167     os_thread_running = to;
00168     /* r0, r1, r18 bis r27, Z und SREG werden hier nach folgender Ueberlegung NICHT gesichert:
00169      * r0, r18 bis r27 und Z wurden bereits vor dem Eintritt in diese Funktion auf dem (korrekten) Stack gesichert!
00170      * Falls wir aus der Timer-ISR kommen, wurden auch r1 und das Statusregister bereits gerettet. Falls nicht,
00171      * duerfen wir das SREG ruhig ueberschreiben, weil der (noch) aktuelle Thread in diesem Fall den Scheduler explizit
00172      * aufrufen wollte, der Compiler also weiss, dass sich das Statusregister aendern kann (kooperativer Threadswitch).
00173      * In r1 muss vor jedem Funktionsaufruf bereits 0 stehen.
00174      */
00175     __asm__ __volatile__(
00176         "in r1, __SREG__        ; r1 == 0           \n\t"
00177         "cli                    ; interrupts off    \n\t"
00178         "push r1                ; push SREG         \n\t"
00179         "push r2                ; save GP registers \n\t"
00180         "push r3                                    \n\t"
00181         "push r4                                    \n\t"
00182         "push r5                                    \n\t"
00183         "push r6                                    \n\t"
00184         "push r7                                    \n\t"
00185         "push r8                                    \n\t"
00186         "push r9                                    \n\t"
00187         "push r10                                   \n\t"
00188         "push r11                                   \n\t"
00189         "push r12                                   \n\t"
00190         "push r13                                   \n\t"
00191         "push r14                                   \n\t"
00192         "push r15                                   \n\t"
00193         "push r16                                   \n\t"
00194         "push r17                                   \n\t"
00195         "push r28                                   \n\t"
00196         "push r29                                   \n\t"
00197     //-- hier ist noch "from" (Z) der aktive Thread     --//
00198         "in r16, __SP_L__   ; switch Stacks         \n\t"
00199         "st Z+, r16                                 \n\t"
00200         "in r16, __SP_H__                           \n\t"
00201         "st Z, r16                                  \n\t"
00202                 //-- live changes here --//
00203         "ld r16, X+                                 \n\t"
00204         "out __SP_L__, r16                          \n\t"
00205         "ld r16, X                                  \n\t"
00206         "out __SP_H__, r16                          \n\t"
00207     //-- jetzt ist schon "to" (X) der aktive Thread     --//
00208         "pop r29            ; restore registers     \n\t"
00209         "pop r28                                    \n\t"
00210         "pop r17                                    \n\t"
00211         "pop r16                                    \n\t"
00212         "pop r15                                    \n\t"
00213         "pop r14                                    \n\t"
00214         "pop r13                                    \n\t"
00215         "pop r12                                    \n\t"
00216         "pop r11                                    \n\t"
00217         "pop r10                                    \n\t"
00218         "pop r9                                     \n\t"
00219         "pop r8                                     \n\t"
00220         "pop r7                                     \n\t"
00221         "pop r6                                     \n\t"
00222         "pop r5                                     \n\t"
00223         "pop r4                                     \n\t"
00224         "pop r3                                     \n\t"

00226         "pop r1                 ; load SREG         \n\t"
00227         "out __SREG__, r1       ; restore SREG      \n\t"
00228         "clr r1                 ; cleanup r1            "

00230         :   "memory"
